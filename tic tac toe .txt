  <script>
    const board = document.getElementById("board");
    const statusText = document.getElementById("status");
    const resetBtn = document.getElementById("reset");
    const newMatchBtn = document.getElementById("newMatch");
    const modeSelect = document.getElementById("mode");
    const scoreX = document.getElementById("scoreX");
    const scoreO = document.getElementById("scoreO");
    const scoreDraw = document.getElementById("scoreDraw");

    let cells = [];
    let currentPlayer = "X";
    let gameActive = true;
    let gameMode = "pvp";

    let scores = { X: 0, O: 0, Draw: 0 };

    const winningPatterns = [
      [0,1,2],
      [3,4,5],
      [6,7,8],
      [0,3,6],
      [1,4,7],
      [2,5,8],
      [0,4,8],
      [2,4,6]
    ];

    function createBoard() {
      board.innerHTML = "";
      cells = [];
      for (let i = 0; i < 9; i++) {
        const cell = document.createElement("div");
        cell.classList.add("cell");
        cell.addEventListener("click", () => handleCellClick(i));
        board.appendChild(cell);
        cells.push(cell);
      }
      currentPlayer = "X";
      statusText.textContent = "Turn: X";
      gameActive = true;
    }

    function handleCellClick(i) {
      if (!gameActive || cells[i].textContent !== "") return;
      cells[i].textContent = currentPlayer;
      checkWinner();

      if (gameMode === "cpu" && gameActive && currentPlayer === "X") {
        currentPlayer = "O";
        statusText.textContent = "Computer's turn...";
        setTimeout(cpuMove, 600);
      } else if (gameActive) {
        currentPlayer = currentPlayer === "X" ? "O" : "X";
        statusText.textContent = "Turn: " + currentPlayer;
      }
    }

    function cpuMove() {
      let available = cells.map((c, i) => (c.textContent === "" ? i : null)).filter(i => i !== null);
      if (available.length === 0) return;

      // Simple AI: try to win, then block, else random
      const move = findBestMove("O") || findBestMove("X") || available[Math.floor(Math.random() * available.length)];
      cells[move].textContent = "O";
      checkWinner();
      if (gameActive) {
        currentPlayer = "X";
        statusText.textContent = "Turn: X";
      }
    }

    function findBestMove(player) {
      for (let pattern of winningPatterns) {
        const [a,b,c] = pattern;
        const values = [cells[a].textContent, cells[b].textContent, cells[c].textContent];
        if (values.filter(v => v === player).length === 2 && values.includes("")) {
          return pattern[values.indexOf("")];
        }
      }
      return null;
    }

    function checkWinner() {
      for (let pattern of winningPatterns) {
        const [a,b,c] = pattern;
        if (cells[a].textContent && cells[a].textContent === cells[b].textContent && cells[a].textContent === cells[c].textContent) {
          cells[a].classList.add("highlight");
          cells[b].classList.add("highlight");
          cells[c].classList.add("highlight");
          statusText.textContent = ${cells[a].textContent} Wins! ðŸŽ‰;
          scores[cells[a].textContent]++;
          updateScores();
          gameActive = false;
          return;
        }
      }

      if (cells.every(c => c.textContent !== "")) {
        statusText.textContent = "It's a Draw!";
        scores.Draw++;
        updateScores();
        gameActive = false;
      }
    }

    function updateScores() {
      scoreX.textContent = scores.X;
      scoreO.textContent = scores.O;
      scoreDraw.textContent = scores.Draw;
    }

    resetBtn.addEventListener("click", createBoard);
    newMatchBtn.addEventListener("click", () => {
      scores = { X: 0, O: 0, Draw: 0 };
      updateScores();
      createBoard();
    });

    modeSelect.addEventListener("change", () => {
      gameMode = modeSelect.value;
      createBoard();
    });

    createBoard();
  </script>
